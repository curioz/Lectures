\section{Syllabus}

\begin{enumerate}[{\bf Lecture 1}]

\begin{singlespace}
\item{Workflow for Computational Fluid Dynamics (CFD) Simulations}
\begin{enumerate}
\item Defining the physical and mathematical problem;
\item Mesh generation and discretisation;
\item Solving the system of equations: linear and parallel processing;
\item Visualisation techniques.
\end{enumerate}
\end{singlespace}

\begin{singlespace}
\item{Solving Partial Differential Equations (PDE)}
\begin{enumerate}
\item Classification;
\item Well-posed problems;
\item Finite difference method.
\end{enumerate}
\end{singlespace}

\begin{singlespace}
\item{Computational Linear Algebra}
\begin{enumerate}%[(i)]
\item Matrices, vectors and vectors spaces;
\item Direct methods (Gauss-Jordan elimination,Pivoting); 
\item Iterative methods (Jacobi and Gauss-Seidel methods, convergence and matrix sparsity).
\end{enumerate}
\end{singlespace}

\begin{comment}
\item{Root-Finding Methods for Nonlinear Equations}
\begin{enumerate}
\item Bisection method;
\item Secant method;
\item Newton's method;
\item Convergence analysis.
\end{enumerate} 
\end{comment}


\end{enumerate}
