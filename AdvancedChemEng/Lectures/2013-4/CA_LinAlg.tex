\begin{enumerate}[{\bf Problem 1:}]
%
\item Generate an algorithm (as pseudo-code) for:
  \begin{enumerate}
    \item LU decomposition {\bf [10 Marks]};
    \item Gauss-Jordan method {\bf [10 Marks]}. 
  \end{enumerate} 
%
\item There are several methods for time-discretisation of differential equations. They can be defined as implicit and explicit methods. Describe the (a) implicit {\it Backward Euler} {\bf 9 Marks]} and (b) explicit {\it Forward Euler} methods {\bf [9 Marks]} for PDEs. Analyse the order of accuracy -- $\mathcal{O}\left(\Delta t\right)^{p}$\;\;\; {\bf [7 Marks]}
%
\item Chemical species are advected (in 1D) at velocity $u_{x}=0.50$ m/s with time-step size $\left(\Delta t\right)$ of $0.5$s through $60$ m. Discretising the transport equation in space and time, the concentration can be calculated, using FDM as,
\begin{displaymath}
\mathcal{C}_{i}^{j+1}=\mathcal{C}_{i}^{j} - \frac{u\Delta t}{\Delta x }\left(\mathcal{C}_{i+1}^{j}-\mathcal{C}_{i}^{j}\right)
\end{displaymath}
where $i\in\left\{1,2,...,N_{x}\right\}$ and $j\in\left\{0,1,...,k\right\}$ are spatial- and time-increment indexes, respectively. Assume:
\begin{itemize}
\item The domain is divided into $N_{x}=5$ nodes;
\item Initial condition is given by,
\[
\mathcal{C}\left(x,t=0\right)=
\left\{\begin{array}{l l}
0.40 & \text{ for }  x \le 8.0 \\
0.25 & \text{ for }  8.0 < x \le 35.0 \\
0.0  & \text{ elsewhere.}
\end{array}\right.\]
\item The {\it ghost-node}: $x_{N_{x}+1}=x_{N_{x}}$
\end{itemize}
Determine $\mathcal{C}_{i}^{j}$ with $j = \left\{1, \cdots, N_{t}\right\}$ $\left(N_{t}=3\right)$ and $i = \left\{1, \cdots, N_{x}\right\}$\;\;\; {\bf [30 Marks]}
%
\item Solve the following system {\bf (showing all steps)} via 
 \begin{displaymath}
             \begin{pmatrix}
                1 & 6 & 2 & 4 \\ 3 & 19 & 4 & 15 \\ 1 & 4 & 8 & -12 \\ 5 & 33 & 9 & 3
             \end{pmatrix}
             \begin{pmatrix}
                x_{1} \\ x_{2} \\ x_{3} \\ x_{4}
             \end{pmatrix}=
             \begin{pmatrix}
                8 \\ 25 \\ 18 \\ 72
             \end{pmatrix}
\end{displaymath}
\begin{enumerate}
\item Gaussian elimination {\bf [5 Marks]};
\item Jacobi {\bf [5 Marks]}.
\item Gauss-Seidel {\bf [5 Marks]};
\item SOR with $\omega=0.4$ {\bf [ 5 Marks]};
\item SOR with $\omega=1.1$ {\bf [ 5 Marks]}.
\end{enumerate}
%
\end{enumerate}
